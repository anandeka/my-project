--Set default Feeding points

insert into SFP_SMELTER_FEEDING_POINT  
select rownum ,'FP-'||substr(PHD.COMPANYNAME,0,INSTRB(PHD.COMPANYNAME||' ' ,' ',1,1)),PHD.PROFILEID ,'Y'  From  PHD_PROFILEHEADERDETAILS  PHD 
WHERE PHD.IS_ACTIVE ='Y' and PHD.PROFILEID IN (select  BPR.PROFILE_ID from BPR_BUSINESS_PARTNER_ROLES  BPR WHERE BPR.ROLE_TYPE_CODE IN ('BUYER','SELLER'));
